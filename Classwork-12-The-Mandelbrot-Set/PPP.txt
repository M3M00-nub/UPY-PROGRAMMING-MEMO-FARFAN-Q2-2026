#INPUT
OPEN file "config.txt" for reading
FOR EACH line IN file
    SPLIT line INTO parameter, value USING "="
    IF value CONTAINS "."
        STORE value AS decimal number IN config[parameter]
    ELSE
        STORE value AS integer number IN config[parameter]
    END IF
END FOR
CLOSE file


DISPLAY config
OPEN file "mandelbrot.csv" for reading
READ ALL lines INTO data_lines
CLOSE file
REMOVE first element FROM data_lines   // discard header row
SET max_iter <- config["max_iter"]
SET width <- config["ancho"]
SET height <- config["alto"]

#PROCESS
CREATE empty image IMG of type HSV with size (width, height)
FOR EACH line IN data_lines
    SPLIT line INTO row, column, iterations USING ","
    CONVERT row, column, iterations TO integer

    IF iterations = max_iter
        SET brightness <- 40
    ELSE
        SET brightness <- INTEGER( (iterations / max_iter) * 255 )
    END IF

    SET PIXEL of IMG AT (column, row) TO (brightness, 255, 255)
END FOR

#OUTPUT
CONVERT IMG FROM HSV TO RGB INTO img_rgb
SAVE img_rgb AS "mandelbrot.png"
DISPLAY "DONE"
